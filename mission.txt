import bpy

# 현재 Armature를 선택한 상태에서 실행하세요
armature = bpy.context.object

# "Mixamo에서 요구하는 이름" : "현재 본 이름" (반대 매핑!)
rename_map = {
    "c_hand_ik_left_pb": "Fist_ik.L",
    "c_hand_ik_right_pb": "Fist_ik.R",
    "c_foot_ik_left_pb": "Foot_ik.L",
    "c_foot_ik_right_pb": "Foot_ik.R",
    # 필요시 더 추가
}

bpy.ops.object.mode_set(mode='EDIT')
edit_bones = armature.data.edit_bones

for mixamo_name, current_name in rename_map.items():
    bone = edit_bones.get(current_name)
    if bone:
        print(f"Renaming '{current_name}' → '{mixamo_name}'")
        bone.name = mixamo_name
    else:
        print(f"Bone '{current_name}' not found!")

bpy.ops.object.mode_set(mode='OBJECT')
print("본 이름 변경 완료!")
